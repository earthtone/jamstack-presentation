## Remote Local Data

```js
const fetch = require('cross-fetch')

module.exports = function (api) {
  api.loadSource(async ({ addCollection }) => {
    try {
      const token = process.env.STORYBLOK_TOKEN
      let req = await fetch(`https://api.storyblok.com/v1/cdn/stories?token=${token}&cv=${Date.now()}&filter_query[component][in]=product`)
      let res = await req.json()

      let contentType = addCollection({
        typeName: 'Product'
      })

      for (const story of res.stories) {
        contentType.addNode({
          ...story,
          id: String(story.id),
          path: `/${story.full_slug}`
        })
      }

      req = await fetch(`https://api.storyblok.com/v1/cdn/stories?token=${token}&cv=${Date.now()}&filter_query[component][in]=gallery-item`)
      res = await req.json()

      contentType = addCollection({
        typeName: 'GalleryItem'
      })

      for (const story of res.stories) {
        contentType.addNode({
          ...story,
          id: String(story.id),
          path: `/${story.full_slug}`
        })
      }
    } catch (err) {
      console.log('ERROR: Unable to fetch CMS Content')
      console.error(err)
    }
  })

  api.createManagedPages(async ({ createPage }) => {
    try {
      const token = process.env.STORYBLOK_TOKEN
      const req = await fetch(`https://api.storyblok.com/v1/cdn/stories?token=${token}&cv=${Date.now()}&filter_query[component][in]=page`)
      const res = await req.json()

      for (const story of res.stories) {
        createPage({
          path: story.full_slug === 'home' ? story.path : `/${story.full_slug}`,
          component: './src/templates/Story.vue',
          context: {
            id: story.id,
            name: story.name,
            content: story.content,
            tags: story.tag_list,
            isDirRoot: story.is_startpage,
            parentId: story.parent_id
          }
        })
      }
    } catch (err) {
      console.log('ERROR: Unable to fetch CMS Content')
      console.error(err)
    }
  })

  api.chainWebpack(config => {
    config.mode('development')
  })
}
```
